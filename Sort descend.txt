ASSUME CS: CODE, DS: DATA
DATA SEGMENT
LIST DB 56H, 12H, 72H, 32H 
COUNT EQU 0004H 
DATA ENDS

CODE SEGMENT
ORG 1000H 
START: MOV AX, DATA
MOV DS, AX
MOV CX, COUNT
MOV DX, CX

AGAIN: MOV SI, OFFSET LIST
MOV CX, DX

BACK: MOV AL, [SI]
INC SI
CMP AL, [SI]
JNC NEXT   ; Jump if not carry, i.e., AL >= [SI]
XCHG [SI], AL
DEC SI
MOV [SI], AL
INC SI

NEXT: LOOP BACK
DEC DX
JNZ AGAIN

MOV AH,4CH
INT 21H 
CODE ENDS
END START
